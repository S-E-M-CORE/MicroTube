287 oatpp::Object<dto::UserTokenPairDTO> isBearerTokenValid(oatpp::String tokenStr)
288 {
289 if (!tokenStr)
290 return nullptr;
291
292 const size_t TokenStart = tokenStr->find_last_of(" ");
293 const oatpp::String Token = tokenStr->substr(TokenStart);
294
295 auto dbResult = isTokenValid(Token);
296 OATPP_ASSERT_HTTP(dbResult->isSuccess(), Status::CODE_500, dbResult->getErrorMessage());
297
298 auto dbResultList = dbResult->fetch<oatpp::List<oatpp::Object<dto::UserTokenPairDTO>>>();
299
300 if (dbResultList->empty())
301 return nullptr;
302
303 OATPP_ASSERT_HTTP(dbResultList->size() == 1, Status::CODE_500, "Internal server error");
304
305 return dbResultList[0];
306 }

308 oatpp::Object<dto::UserTokenPairDTO> checkBearerThrowIfInvalid(oatpp::String tokenStr)
309 {
310 OATPP_ASSERT_HTTP(tokenStr, Status::CODE_400, "Bearer required");
311
312 const size_t TokenStart = tokenStr->find_last_of(" ");
313 const oatpp::String Token = tokenStr->substr(TokenStart + 1);
314
315 auto dbResult = isTokenValid(Token);
316 OATPP_ASSERT_HTTP(dbResult->isSuccess(), Status::CODE_500, dbResult->getErrorMessage());
317
318 auto dbResultList = dbResult->fetch<oatpp::List<oatpp::Object<dto::UserTokenPairDTO>>>();
319
320 OATPP_ASSERT_HTTP(dbResultList->size() == 1, Status::CODE_500, "Internal server error");
321 OATPP_ASSERT_HTTP(!dbResultList->empty(), Status::CODE_404, "Unknown token");
322
323 return dbResultList[0];
324 }
