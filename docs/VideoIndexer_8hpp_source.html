88 if (instance->m_cv.wait_for(lock, std::chrono::minutes(IndexIntervallMinutes), [&end] { return end.load(); }))

340 {
341 std::string searchStr = str;
342 std::transform(searchStr.begin(), searchStr.end(), searchStr.begin(), ::tolower);
343
344 size_t pos = 0;
345 while ((pos = searchStr.find('+', pos)) != std::string::npos)
346 {
347 searchStr.replace(pos, 1, " ");
348 ++pos;
349 }
350
351 pos = 0;
352 while ((pos = searchStr.find("%20", pos)) != std::string::npos)
353 {
354 searchStr.replace(pos, 3, " ");

356 }

359 auto searchIn = [&searchStr](const oatpp::String& text) -> bool
360 {
361 if (!text || text->size() == 0)
362 return false;
363
364 std::string textLower = text;
365 std::transform(textLower.begin(), textLower.end(), textLower.begin(), ::tolower);
366
367 return textLower.find(searchStr) != std::string::npos;
368 };
