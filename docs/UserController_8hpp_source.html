240 static inline oatpp::String generateBearerToken(const size_t length = 50)
241 {
242 const std::string characters{ "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789" };
243 std::random_device rd;
244 std::mt19937 generator(rd());
245 std::uniform_int_distribution<size_t> distribution(0, characters.size() - 1);
246
247 std::string token{};
248
249 for (size_t i = 0; i < length; ++i)
250 token += characters[distribution(generator)];

252 return token;
